


CREATE TABLE Company (
    Id          VARCHAR PRIMARY KEY,
    CompanyName VARCHAR,
    City        VARCHAR,
    Address     VARCHAR,
    PhoneNumber VARCHAR,
    Status      VARCHAR,
    FirstEntryDate        VARCHAR,
    Details     VARCHAR
);

CREATE TABLE `User` (
	`Username`	VARCHAR,
	`Password`	VARCHAR,
	`ID`	VARCHAR,
	PRIMARY KEY(`ID`)
);
